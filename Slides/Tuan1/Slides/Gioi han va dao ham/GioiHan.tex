\begin{frame}{Giới hạn}
    \begin{tcolorbox}[colback=blue!10, colframe=blue!50!black, title=Định nghĩa]
    Giới hạn của hàm số \(f(x)\) tại điểm \(x=a\) được ký hiệu là:
    \begin{equation}
        \lim_{x \to a} f(x) = L
    \end{equation}
    là giá trị mà hàm số tiến tới khi \(x\) tiến tới \(a\).
    \end{tcolorbox}
\end{frame}
